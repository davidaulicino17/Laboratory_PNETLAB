! zerotouch disable

enable
configure terminal

hostname Leaf-1
no aaa root

! Configurazione Globale
system mtu 9214
ip routing

! Interfacce Underlay (OSPF)
interface Loopback0
   ip address 1.1.1.11/32
   
interface Ethernet1
   description Link-to-Spine-1
   no switchport
   mtu 9000
   ip address 10.0.1.1/31
   ip ospf network point-to-point

interface Ethernet2
   description Link-to-Spine-2
   no switchport
   mtu 9000
   ip address 10.0.5.1/31
   ip ospf network point-to-point

! Interfaccia Access (Host)
interface Ethernet5
   description Link-to-Server-1.1
   switchport
   switchport access vlan 10

! Routing Underlay (OSPF)
router ospf 1
   router-id 1.1.1.11
   network 1.1.1.0/24 area 0.0.0.0
   network 10.0.0.0/16 area 0.0.0.0

! Configurazione VXLAN
interface Vxlan1
   description VTEP Interface
   vxlan source-interface Loopback0
   vxlan udp-port 4789
   vxlan vlan 10 vni 10010

! Configurazione Tenant (L2VNI + DAG)
vlan 10
   name TENANT_BLU
   
interface Vlan10
   description DAG Tenant BLU
   ip address virtual 192.168.10.1/24

! Routing Overlay (BGP Client)
router bgp 65000
   router-id 1.1.1.11
   neighbor SPINE_PEER_GROUP peer group
   neighbor SPINE_PEER_GROUP remote-as 65000
   neighbor SPINE_PEER_GROUP update-source Loopback0
   neighbor 1.1.1.1 peer group SPINE_PEER_GROUP
   neighbor 1.1.1.2 peer group SPINE_PEER_GROUP
   !
   address-family evpn
      neighbor SPINE_PEER_GROUP activate
   !
   vlan 10
      rd auto
      route-target both 10010:10010
      redistribute learned

end
